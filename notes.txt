CONSULTA SQL:

    SELECT
    cliente.codigo_cliente,
    CONCAT(cliente.nombres, '', cliente.apellidos) NombreCliente, 
    cliente.idtipocliente,
    tipo_clientes.nombre_tipo,
    facturacion.idfacturacion, 
    facturacion.idcliente, 
    facturacion.monto_facturacion,
    DB_NAME() country
    FROM facturacion
    INNER JOIN cliente ON facturacion.idcliente = cliente.idcliente 
    INNER JOIN tipo_clientes ON cliente.idtipocliente = tipo_clientes.idtipocliente

    country
